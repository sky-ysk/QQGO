func (h *Hub) handleChatMessage(c *ws.Conn, msg *model.Message) {
	msg.FromUID = c.UID

	if _, err := h.svc.GetUserByUID(msg.ToUID); err != nil {
		log.Printf("[chat] target user %s not found", msg.ToUID)
		c.WriteJSON(&model.Message{
			MsgType:   model.MsgTypeServerAck,
			ID:        -1,
			ClientSeq: msg.ClientSeq,
			Content:   "user not found",
		})
		return
	}

	if err := h.svc.HandleMessage(context.Background(), msg); err != nil {
		log.Printf("[chat] store error: %v", err)
		c.WriteJSON(&model.Message{
			MsgType: model.MsgTypeServerAck,
			ID:      -1,
			Content: "store failed",
		})
		return
	}

	c.WriteJSON(&model.Message{
		MsgType:   model.MsgTypeServerAck,
		ID:        msg.ID,
		ClientSeq: msg.ClientSeq,
		Content:   "ok",
	})

	if msg.GroupID != "" {
		h.broadcastToGroup(msg)
	} else {
		h.sendToUser(msg.ToUID, msg)
	}
}
```

var ErrGroupNotFound = errors.New("group not found")
var ErrGroupNotEmpty = errors.New("group is not empty")

func (s *ChatService) CreateFriendGroup(uid, name string) error {
	if name == "" || name == "待处理" || name == "我的好友" {
		return errors.New("invalid group name")
	}
	var count int64
	s.db.Model(&model.FriendGroup{}).Where("uid = ? AND group_name = ?", uid, name).Count(&count)
	if count > 0 {
		return errors.New("group already exists")
	}
	fg := &model.FriendGroup{UID: uid, GroupName: name}
	return s.db.Create(fg).Error
}

func (s *ChatService) DeleteFriendGroup(uid, name string) error {
	if name == "我的好友" {
		return errors.New("cannot delete default group")
	}
	var count int64
	s.db.Model(&model.Friend{}).Where("uid = ? AND group_name = ? AND status = ?", uid, name, model.FriendStatusAccepted).Count(&count)
	if count > 0 {
		return ErrGroupNotEmpty
	}
	result := s.db.Where("uid = ? AND group_name = ?", uid, name).Delete(&model.FriendGroup{})
	if result.RowsAffected == 0 {
		return ErrGroupNotFound
	}
	return nil
}
```

func (s *ChatService) MoveFriendGroup(uid string, friendQQNumber int64, groupName string) error {
	friendUser, err := s.GetUserByQQNumber(friendQQNumber)
	if err != nil {
		return ErrUserNotFound
	}

	if groupName != "我的好友" {
		var count int64
		s.db.Model(&model.FriendGroup{}).Where("uid = ? AND group_name = ?", uid, groupName).Count(&count)
		if count == 0 {
			return ErrGroupNotFound
		}
	}

	result := s.db.Model(&model.Friend{}).
		Where("uid = ? AND friend_uid = ? AND status = ?", uid, friendUser.UID, model.FriendStatusAccepted).
		Update("group_name", groupName)
	if result.RowsAffected == 0 {
		return ErrNotFriend
	}
	return nil
}
```

func (h *Hub) handleFriendCreateGroup(c *ws.Conn, msg *model.Message) {
	if c.UID == "" {
		h.writeFriendError(c, "not logged in")
		return
	}

	if err := h.svc.CreateFriendGroup(c.UID, msg.Content); err != nil {
		h.writeFriendError(c, err.Error())
		return
	}

	log.Printf("[friend] group created: uid=%s name=%s", c.UID, msg.Content)
	h.writeFriendResult(c, "group created")
}

func (h *Hub) handleFriendDeleteGroup(c *ws.Conn, msg *model.Message) {
	if c.UID == "" {
		h.writeFriendError(c, "not logged in")
		return
	}

	if err := h.svc.DeleteFriendGroup(c.UID, msg.Content); err != nil {
		h.writeFriendError(c, err.Error())
		return
	}

	log.Printf("[friend] group deleted: uid=%s name=%s", c.UID, msg.Content)
	h.writeFriendResult(c, "group deleted")
}
```
